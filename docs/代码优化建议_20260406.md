### 🔴 2. Locator 类闭包变量丢失

**位置**: `conftest.py:642-797`

**问题代码**:
```python
def __getattribute__(self, attr):
    # ...
    def wrapped(*args, **kwargs):
        step_title = None
        if attr == "_sync" and self.__last_step:  # 闭包中 self 可能丢失
            step_title = self.__last_step
        # ...
```

**风险**: 在复杂场景下，闭包中的 `self` 引用可能指向错误的对象

**建议修复**: 使用 `functools.wraps` 并保持原始引用
```python
import functools

def __getattribute__(self, attr):
    orig_attr = super().__getattribute__(attr)
    if callable(orig_attr):
        @functools.wraps(orig_attr)
        def wrapped(*args, **kwargs):
            # 在函数开始时获取 self，避免闭包问题
            current_self = self
            # ...
        return wrapped
    return orig_attr
```

### 🔴 3. 全局状态线程不安全

**位置**: `utils/globalMap.py`

```python
# ❌ 当前代码 - 类变量在多线程下不安全
class GlobalMap:
    my_dict = {}  # 类变量，所有实例共享
    
    def set(self, key, value):
        self.my_dict[key] = value
```

**风险**: 启用多线程测试时（`pytest-xdist`），数据可能错乱

**建议修复**:
```python
# ✅ 使用 threading.local() 实现线程隔离
import threading

class GlobalMap:
    def __init__(self):
        self._local = threading.local()
    
    def _get_dict(self):
        if not hasattr(self._local, 'my_dict'):
            self._local.my_dict = {}
        return self._local.my_dict
    
    def set(self, key, value):
        self._get_dict()[key] = value
    
    def get(self, key):
        return self._get_dict().get(key)
    
    def delete(self, key):
        self._get_dict().pop(key, None)

# 或者更简单：直接使用 pytest fixture 的作用域管理
```

### 🟡 6. BasePage 职责过重

**位置**: `module/BasePage.py`（194 行）

**问题**: `PageObject` 类包含了太多 UI 组件操作：
- 表单输入 (`el_input`)
- 下拉框 (`el_select`)
- 单选框 (`el_radio`)
- 开关 (`el_switch`)
- 日期选择 (`el_datetime`)

- Hover 重试 (`hover_retry`)

**建议重构**: 拆分为独立 Helper 类
```python
# module/helpers/form_helper.py
class FormHelper:

        self.page = page
    
    def fill_input(self, label: str, value: str):
        # ...
    
    def select_option(self, label: str, value: str):
        # ...

# module/helpers/table_helper.py
class TableHelper:
    def __init__(self, page: Page):
        self.page = page
    
    def get_cell(self, table_locator, row, col):
        # ...

# 然后在 PageObject 中组合使用
class PageObject:

        self.page = page
        self.form = FormHelper(page)

"""
测试数据工厂 - 用于生成各类测试数据
"""
import random
import string
from datetime import datetime, timedelta


class DataFactory:
    """测试数据生成器"""
    
    @staticmethod
    def random_username(prefix="user", length=8) -> str:
        """生成随机用户名"""
        suffix = ''.join(random.choices(string.ascii_lowercase + string.digits, k=length))
        return f"{prefix}_{suffix}"
    
    @staticmethod
    def random_phone() -> str:
        """生成随机中国大陆手机号"""
        prefixes = ['138', '139', '150', '151', '152', '158', '159', '188']
        return random.choice(prefixes) + ''.join(random.choices(string.digits, k=8))
    
    @staticmethod
    def random_email(domain="test.com") -> str:
        """生成随机邮箱"""
        username = ''.join(random.choices(string.ascii_lowercase + string.digits, k=10))
        return f"{username}@{domain}"
    
    @staticmethod
    def random_id_card() -> str:
        """生成随机身份证号（符合格式规则）"""
        # 简化版本，实际使用建议用 faker 库
        region = random.choice(['110101', '310101', '440101'])  # 北京/上海/广州
        birth = datetime.now().replace(year=random.randint(1980, 2005)).strftime('%Y%m%d')
        seq = ''.join(random.choices(string.digits, k=3))
        check = random.choice(string.digits + 'Xx')
        return f"{region}{birth}{seq}{check}"
    
    @staticmethod
    def random_order_id(prefix="ORD") -> str:
        """生成随机订单号"""
        timestamp = datetime.now().strftime('%Y%m%d%H%M%S')
        suffix = ''.join(random.choices(string.digits, k=4))
        return f"{prefix}{timestamp}{suffix}"
    
    @staticmethod
    def random_date(start_days=-365, end_days=365, fmt="%Y-%m-%d") -> str:
        """生成随机日期"""
        delta = random.randint(start_days, end_days)
        date = datetime.now() + timedelta(days=delta)
        return date.strftime(fmt)

# 新增 utils/env_detector.py
from enum import Enum

class EnvironmentType(Enum):
    LOCAL = "local"
    JENKINS = "jenkins"
    GITLAB_CI = "gitlab_ci"
    GITHUB_ACTIONS = "github_actions"
    DEV = "dev"
    TEST = "test"
    PROD = "prod"


def detect_environment() -> dict:
    """
    自动检测运行环境
    Returns: 包含环境信息的字典
    """
    indicators = {
        EnvironmentType.JENKINS: ['JENKINS_URL', 'BUILD_NUMBER', 'WORKSPACE'],
        EnvironmentType.GITLAB_CI: ['GITLAB_CI', 'CI_PIPELINE_ID'],
        EnvironmentType.GITHUB_ACTIONS: ['GITHUB_ACTIONS', 'GITHUB_RUN_ID'],
    }
    
    for env_type, env_vars in indicators.items():
        if any(os.getenv(var) for var in env_vars):
            return {"type": env_type.value, "raw": {var: os.getenv(var) for var in env_vars}}
    
    return {"type": EnvironmentType.LOCAL.value}
```
